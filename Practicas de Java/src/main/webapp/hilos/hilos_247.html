Realizar N tareas en paralelo

Hay algunos casos en los que las tareas que se realizan en paralelo pueden ser útiles. Esta opción es especialmente atractiva cuando:
  • las tareas son independientes unas de otras, poer hacen esencialmente la misma cosa
  • las tareas están limitadas por operaciones lentas - habitualmente, acceso a la red o acceso al disco

Un buen ejemplo de esto viene de la administración de un sitio web. Los administradoers necesitan monitorizar un conjunto de N sitios web, así que ellos podrían enviar un 'ping' de alguna clase a cada sitio, para ver si se devuelve una respuesta saludable en cada caso. Tenga en cuenta que estos pings se ajustan a los criterios anteriores: cada ping es independiente de los demás y, sin embargo, es esencialmente la misma tarea.

Abajo hay una implementación de ejemplo de esta idea. Para el propósito de la comparación, este ejemplo en realidad tiene 3 diferentes implementaciones:

  • ejecutar tareas en paralelo, e informar el resultado de cada tarea tan pronto como esta se complet
  • ejecutar tareas en paralelo, pero informar de los resultados de todas las tareas sólo al final , después de que todas ellas se hayan completado.
  • ejecutar las tareas en serie, una después de otra, y todas dentro de un único hilo; informar de cada tarea cuando esta se completa.
Puede comparar el rendimiento de estas 3 implementaciones, para ver que trabaja mejor para usted. 
import java.io.IOException;
import java.net.MalformedURLException;
import java.net.URL;
import java.net.URLConnection;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collection;
import java.util.List;
import java.util.Objects;
import java.util.concurrent.Callable;
import java.util.concurrent.CompletionService;
import java.util.concurrent.ExecutionException;
import java.util.concurrent.ExecutorCompletionService;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;
import java.util.concurrent.Future;

/** 
  Ping N sitios web en paralelo.
 El ping simplemente hace un GET, y mira en la primera línea de cabecera.
 Este ejemplo podría ser aplicado a muchas clases de tareas similares.
 Aquí no se utiliza ningún tiempo muerto. Como de costumbre, tenga cuidado con el calentamiento.
  del compilador just-in-time. Es posible que desee utilizar -Xint.
*/
public final class CheckSites {
  
  /** Ejecuta esta herramienta. */
  public static final void main(String... aArgs) {
    CheckSites checker = new CheckSites();
    try {
      log("Paralelamente, informe cada uno a medida que se complete:");
      checker.pingAndReportEachWhenKnown();
      
      log("Paralelamente, informe todo al final:");
      checker.pingAndReportAllAtEnd();
      
      log("Secuencial, informa cada uno a medida que se complete:");
      checker.pingAndReportSequentially();
    }
    catch(InterruptedException ex){
      Thread.currentThread().interrupt();
    }
    catch(ExecutionException ex){
      log("Problema ejecutando el worker: " + ex.getCause());
    }
    catch(MalformedURLException ex){
      log("URL Errónea : " + ex.getCause());
    }
    log("Done.");
  }

  /** 
    Comprueba N sitios, en paralelo, usando hasta 4 hilos.
    Informa del resultado de cada 'ping' a medida que ingresa.
   (Este es probablemente el estilo que la mayoría preferiría.)
  */ 
  void pingAndReportEachWhenKnown()  throws InterruptedException, ExecutionException  {
    int numThreads = URLs.size() > 4 ? 4 : URLs.size(); //máximo 4 hilos
    ExecutorService executor = Executors.newFixedThreadPool(numThreads);
    CompletionService<PingResult> compService = new ExecutorCompletionService<>(executor);
    for(String url : URLs){
      Task task = new Task(url);
      compService.submit(task);
    }
    for(String url : URLs){
      Future<PingResult> future = compService.take();
      log(future.get());
    }
    executor.shutdown(); //siempre reclamar los recursos
  }
  
  /** 
   Comrprueba N sitios, en paralelo, usando hasta 4 hilos.
   Informe los resultados solo cuando todos hayan finalizado.
  */ 
  void pingAndReportAllAtEnd() throws InterruptedException, ExecutionException {
    Collection<Callable<PingResult>> tasks = new ArrayList<>();
    for(String url : URLs){
      tasks.add(new Task(url));
    }
    int numThreads = URLs.size() > 4 ? 4 : URLs.size(); //max 4 threads
    ExecutorService executor = Executors.newFixedThreadPool(numThreads);
    List<Future<PingResult>> results = executor.invokeAll(tasks);
    for(Future<PingResult> result : results){
      PingResult pingResult = result.get();
      log(pingResult);
    }
    executor.shutdown(); //siempre reclamar los recursos
  }

  /**
   Comprueba N sitios, pero secuencialmente, no en paralelo.
   No utiliza varios subprocesos en absoluto.
  */
  void pingAndReportSequentially() throws MalformedURLException {
    for(String url : URLs){
      PingResult pingResult = pingAndReportStatus(url);
      log(pingResult);
    }
  }
  
  // PRIVADO
  
  private static final List<String> URLs = Arrays.asList(
    "http://www.youtube.com/", "http://www.google.ca/", 
    "http://www.date4j.net", "http://www.web4j.com"
  );
  
  private static void log(Object aMsg){
    System.out.println(Objects.toString(aMsg));
  }
  
  /** Intenta hacer ping a una URL. Devuelve verdadero solo si tiene exito. */
  private final class Task implements Callable<PingResult> {
    Task(String aURL){
      fURL = aURL;
    }
    /** Accede a una URL, y ve si tiene una respuesta saludable */
    @Override public PingResult call() throws Exception {
      return pingAndReportStatus(fURL);
    }
    private final String fURL;
  }
  
  private PingResult pingAndReportStatus(String aURL) throws MalformedURLException {
    PingResult result = new PingResult();
    result.URL = aURL;
    long start = System.currentTimeMillis();
    URL url = new URL(aURL);
    try {
      URLConnection connection = url.openConnection();
      int FIRST_LINE = 0;
      String firstLine = connection.getHeaderField(FIRST_LINE);
      result.SUCCESS = true;
      long end = System.currentTimeMillis();
      result.TIMING = end - start;
    }
    catch(IOException ex){
      //ignorar - falla.
    }
    return result;
  }

  /** Estructura simple que contiene todos los datos relacionados con un ping. */
  private static final class PingResult {
    String URL;
    Boolean SUCCESS;
    Long TIMING;
    @Override public String toString(){
      return "Resultado:" + SUCCESS + " " + TIMING + " msecs " + URL;
    }
  }
}
