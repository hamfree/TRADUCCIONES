Compile las expresiones regulares una vez

Los objetos Pattern compilan las expresiones regulares que se le pasan. (La compilación ocurre en memoria). Si una expresión regular se usa muchas veces, entonces esta compilación debe ser realizada solo una vez.

Tenga cuidado con estos métodos de conveniencia, que compilan una expresión regular cada vez que se les llama:

  • Pattern.matches
  • String.matches
  • String.replaceAll
  • String.replaceFirst
  • String.split
Ejemplo

Aquí tiene una comparación del rendimiento de String.matches contra el uso de Pattern y Matcher.

Un ejemplo de ejecución (con el compilador HotSpot desactivado) da: 

Tiempo para String: 67.693 ms
Tiempo para Pattern: 12.178 ms
    

import java.io.IOException;
import java.nio.charset.Charset;
import java.nio.charset.StandardCharsets;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.util.Scanner;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

/** KDJ 7+. */
public final class FindMatch {

  /**
   Cuenta el numero de veces que ocurre la palabra "assert" en un fichero.
  
   Compara los tiempos de ejecución de las dos técnicas.
  */
  public static void main (String... aArguments) throws IOException{
    FindMatch findMatch = new FindMatch("assert");
    Path path = Paths.get("C:\\Temp\\InheritanceInterpreter.java");

    Stopwatch stopwatch = new Stopwatch();
    stopwatch.start();
    findMatch.countMatchesUsingString(path);
    stopwatch.stop();
    log("Tiempo para String: " + stopwatch);

    stopwatch.start();
    findMatch.countMatchesUsingPattern(path);
    stopwatch.stop();
    log("Tiempo para Pattern: " + stopwatch);
  }

  public FindMatch(String aTarget){
    fTargetWord = aTarget;
    fPattern = Pattern.compile(fTargetWord);
  }

  int countMatchesUsingString(Path aPath) throws IOException {
    int result = 0;
    try (Scanner scanner =  new Scanner(aPath, ENCODING.name())){
      while (scanner.hasNextLine()){
        String line = scanner.nextLine();
        if( line.matches(fTargetWord) ){
          ++ result;
        }
      }      
    }
    return result;
  }
  
  int countMatchesUsingPattern(Path aPath) throws IOException {
    int result = 0;
    Matcher matcher = fPattern.matcher("");
    try (Scanner scanner =  new Scanner(aPath, ENCODING.name())){
      while (scanner.hasNextLine()){
        String line = scanner.nextLine();
        matcher.reset(line); //reinicia la entrada cada vez
        if (! matcher.find()) {
          ++result;
        }
      }      
    }
    return result;
  }
  
  // PRIVADO
  private final String fTargetWord;
  private final Pattern fPattern;
  private final static Charset ENCODING = StandardCharsets.UTF_8;

  private static void log(Object aObject){
    System.out.println(aObject);
  }
}
