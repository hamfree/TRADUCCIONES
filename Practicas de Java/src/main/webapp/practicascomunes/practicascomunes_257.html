Tenga cuidado de las Marcas de Orden de Byte

Cuando trata con archivos de texto que tienen codificación Unicode, algunas herramientas antepondrán un carácter especial llamado una marca de orden de bytes (BOM, en inglés) al archivo. El BOM es de 2 a 4 bytes de largo, de acuerdo a la codificación. (En la codificación UTF-8, por ejemplo, la marca de orden de byes es de 3 bytes de longitud). El BOM no es habitualmente representado de forma visual en un editor, así que puede ser difícil de detectar.

Probablemente la codificación Unicode más ampliamente usada sea la codificación UTF-8. Sin embargo, en UTF-8, no se requiere la BOM o no es incluso recomendado por el estándar de UTF-8. Cuando un archivo de texto que tiene codificación UTF-8 está bajo su control, habitualmente debe asegurarse de que no se dalva con una marca de orden de bytes, porque puede conducir a problemas. (Siempre debe ser seguro eliminarla).

Esto es particularmente importante en el contexto de una aplicación web. El HTML, las Java Server Pages, los archivos de etiquetas, etc., por lo general, deben publicarse con codificación UTF-8, sin usar una BOM. Si no, el usuario final puede ver caracteres de extraño aspecto en su navegador, o quizás, líneas vacías extra y no deseadas. Puede también causar que no funcionen bien las hojas de estilo en cascada.

Desafortunadamente, en el sistema operativo Windows, la herramienta Bloc de Notas siempre agregará una marca de orden de bytes cuando salva archivos de texto como UTF-8, así que necesita proceder con cuidado cuando use el Bloc de Notas.

Algunos editores de texto le darán información sobre la presencia o ausencia de una BOM, y algunos otros no lo harán. Para ayudarle a controlar su código fuente, aquí tiene una clase que detectará y opcionalmente eliminará las BOMs UTF-8 de un árbol de fuentes: 

import java.io.BufferedInputStream;
import java.io.BufferedOutputStream;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

/**
 Detecta y elimina Marcas de Orden de Bytes (BOMs) de archivos de texto salvados con 
 codificación Unicode.

 <P>Herramienta solo de desarrollo. Si usa esta herramienta para eliminar BOMs, por favor asegúrese de que ha hecho una 
 copia de seguridad.
 
 <P>Esta clase asume la codificación UTF-8 para la BOM, pero se cambia facilmente para manejar cualquier codificación. Vea  
 http://en.wikipedia.org/wiki/Byte_order_mark para más información.
 KDJ 5+.
*/ 
public final class BomDetector {
  
  /** Ejecute la herramienta contra un directorio raíz.*/
  public static void main(String... aArgs) throws IOException{
    BomDetector bom = new BomDetector(
     "C:\\Temp3\\test\\", 
     ".txt", ".jsp", ".jspf", ".tag", ".html", 
     ".css", ".xml", ".js", ".sql", ".tld"
    );
  
    int count = 0;
    for(String file : bom.findBOMs()){
      log(file);
      ++count;
    }
    log("Numero de ficheros con  BOM:" + count);
    
    /*
    for(String file : bom.removeBOMs()){
      log("Eliminada BOM de: " + file);
    }
    */
  }
  
  public BomDetector(String aRootDirectory, String... aFileExtensions){
    fRootDir = new File(aRootDirectory);
    fExtensions = Arrays.asList(aFileExtensions);
    if(!fRootDir.exists() || fRootDir.isFile() ){
      throw new RuntimeException("Directorio raíz no válido.");
    }
  }
  
  /** Encuentra archivos con BOMs bajo el directorio raiz dado. Devuelve sus nombres. */
  public List<String> findBOMs() throws IOException {
    List<String> result = new ArrayList<String>();
    for(File textFile : findTextFilesBeneath(fRootDir)){
      if(startsWithBOM(textFile)){
        result.add(textFile.getCanonicalPath());
      }
    }
    return result;
  }
  
  /** 
   Encuentra y elimina BOMs de archivos bajo el directorio raiz dado.
   Sobreescribe los archivos.
   Devuelve los nombres de los archvios afectados.
  */
  public List<String> removeBOMs() throws IOException{
    List<String> result = new ArrayList<String>();
    for(String bomFile : findBOMs()){
      stripBomFrom(bomFile);
      result.add(bomFile);
    }
    return result;
  }
  
  // PRIVADO
  private File fRootDir;
  private List<String> fExtensions;
  
  /** Codificaciones diferentes tendrán diferentes BOMs. Esta es para UTF-8. */
  private final int[] BYTE_ORDER_MARK = {239, 187, 191};
  
  private static void log(Object aThing){
    System.out.println(String.valueOf(aThing));
  }
  
  private List<File> findTextFilesBeneath(File aStartingDir) throws IOException {
    List<File> result = new ArrayList<File>();
    File[] filesAndDirs = aStartingDir.listFiles();
    List<File> filesDirs = Arrays.asList(filesAndDirs);
    for(File file : filesDirs){
      if (isTextFile(file)){
        result.add(file);
      }
      if( file.isDirectory() ) {
        //¡¡llamada recursiva!!
        List<File> deeperList = findTextFilesBeneath(file);
        result.addAll(deeperList);
      }
    }
    return result;
  }
  
  private boolean isTextFile(File aFile) throws IOException{
    boolean result = false;
    String fileName = aFile.getCanonicalPath();
    int finalDot = fileName.lastIndexOf(".");
    if (finalDot > -1){
      String extension = fileName.substring(finalDot);
      result = fExtensions.contains(extension);
    }
    return result;
  }
  
  private boolean startsWithBOM(File aTextFile) throws IOException {
    boolean result = false;
    if(aTextFile.length() < BYTE_ORDER_MARK.length) return false;
    //abre como bytes aquí, no caracteres
    int[] firstFewBytes = new int[BYTE_ORDER_MARK.length];
    InputStream input = null;
    try {
      input = new FileInputStream(aTextFile);
      for(int index = 0; index < BYTE_ORDER_MARK.length; ++index){
        firstFewBytes[index] = input.read(); //lee un único byte
      }
      result = Arrays.equals(firstFewBytes, BYTE_ORDER_MARK);
    }
    finally {
      input.close();
    }
    return result;
  }
  
  private void stripBomFrom(String aTextFile) throws IOException{
    File bomFile = new File(aTextFile);
    long initialSize = bomFile.length();
    long truncatedSize = initialSize - BYTE_ORDER_MARK.length;
    byte[] memory = new byte[(int)(truncatedSize)];
    InputStream input = null;
    try {
      input = new BufferedInputStream(new FileInputStream(bomFile));
      input.skip(BYTE_ORDER_MARK.length);
      int totalBytesReadIntoMemory = 0;
      while(totalBytesReadIntoMemory < truncatedSize){
        int bytesRemaining = (int)truncatedSize - totalBytesReadIntoMemory;
        int bytesRead = input.read(memory, totalBytesReadIntoMemory, bytesRemaining);
        if(bytesRead > 0){
          totalBytesReadIntoMemory = totalBytesReadIntoMemory + bytesRead;
        }
      }
      overwriteWithoutBOM(memory, bomFile);
    }
    finally {
      input.close();
    }
    File after = new File(aTextFile);
    long finalSize = after.length();
    long changeInSize = initialSize - finalSize;
    if(changeInSize != BYTE_ORDER_MARK.length){
      throw new RuntimeException(
        "Cambio de tamaño de archivo: " + changeInSize + 
        " Cambio esperado: " + BYTE_ORDER_MARK.length  
      );
    }
  }
  
  private void overwriteWithoutBOM(
    byte[] aBytesWithoutBOM, File aTextFile
  ) throws IOException{
    OutputStream output = null;
    try {
      output = new BufferedOutputStream(new FileOutputStream(aTextFile));
      output.write(aBytesWithoutBOM);
    }
    finally {
      output.close();
    }
  }
}
