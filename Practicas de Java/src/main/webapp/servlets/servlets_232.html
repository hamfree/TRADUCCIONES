Servir contenido binario

Los servlets habitualmente sirven alguna forma de texto - HTML, XML, datos JSON, texto plano, y así. Los Servlets también pueden servir contenido binario, tales como imágenes o archivos .pdf.

Aquí tiene un ejemplo de un servlet simple que sirve contenido binario. Sirve un archivo .pdf de un nombre fijo, ubicado en el directorio raíz de una aplicación web.

import java.io.*;

import javax.servlet.ServletException;
import javax.servlet.ServletOutputStream;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

/**
 Servlet que sirve datos binarios, no texto.
  
 <P>Este servlet sirve un fichero .pdf estático, pero la idea es la misma con otros
 tipos de datos.
*/
public class ServeBinary extends HttpServlet {

  @Override protected void doGet(
    HttpServletRequest aRequest, HttpServletResponse aResponse
  ) throws ServletException, IOException {
    serveBinary(aRequest, aResponse);
  }
   
  @Override protected void doPost(
    HttpServletRequest aRequest, HttpServletResponse aResponse
  ) throws ServletException, IOException {
    serveBinary(aRequest, aResponse);
  }
  
  // PRIVADO

  //omitido...
  
  private void serveBinary(
    HttpServletRequest aRequest, HttpServletResponse aResponse
  ) throws IOException {
    log("servlet ServeBinary.");
   
    //IMPORTANTE: establece el tipo MIME del flujo de respuesta 
    aResponse.setContentType("application/pdf");
    
    //sirve un archivo fijo, ubicado en la carpeta raíz de esta aplicación web
    try (
      ServletOutputStream output = aResponse.getOutputStream();
      InputStream input = getServletContext().getResourceAsStream("test.pdf");
    ){ 
      //transfiere el flujo de entrada a un flujo de salida, a través de un buffer
      byte[] buffer = new byte[2048];
      int bytesRead;    
      while ((bytesRead = input.read(buffer)) != -1) {
         output.write(buffer, 0, bytesRead);
      }
    }
    
    log("Hecho.");
  }
}
