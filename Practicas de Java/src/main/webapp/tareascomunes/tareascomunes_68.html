Listado de ficheros recursivos

Usando el KDJ 7+, un listado recursivo de archivos es una tarea relativamente simple. Puede habitualmente usar SimpleFileVisitor. 
import java.io.IOException;
import java.nio.file.FileVisitResult;
import java.nio.file.FileVisitor;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.nio.file.SimpleFileVisitor;
import java.nio.file.attribute.BasicFileAttributes;

/** Listado recursivo con SimpleFileVisitor en el KDJ 7. */
public final class FileListingVisitor {
  
  public static void main(String... aArgs) throws IOException{
    String ROOT = "C:\\test";
    FileVisitor<Path> fileProcessor = new ProcessFile();
    Files.walkFileTree(Paths.get(ROOT), fileProcessor);
  }

  private static final class ProcessFile extends SimpleFileVisitor<Path> {
    @Override public FileVisitResult visitFile(
      Path aFile, BasicFileAttributes aAttrs
    ) throws IOException {
      System.out.println("Procesando archivo:" + aFile);
      return FileVisitResult.CONTINUE;
    }
    
    @Override  public FileVisitResult preVisitDirectory(
      Path aDir, BasicFileAttributes aAttrs
    ) throws IOException {
      System.out.println("Procesando directorio:" + aDir);
      return FileVisitResult.CONTINUE;
    }
  }
}
Estilo más antiguo, KDJ 6-

La clase File tiene métodos para listar los contenidos de un directorio único, tal como list y listFiles, pero no hay método para listar todos los archivos en un árbol de directorios.

Aquí tiene una utilidad simple que devuelve una List ordenada de objetos File que reside bajo un directorio raíz dado. Usa la recursión. 

import java.util.*;
import java.io.*;

/**
* Listado de archivos recursivos en un directorio específico.
*  
* @author javapractices.com
* @author Alex Wong
* @author usuario anónimo
*/
public final class FileListing {

  /**
  * Demuestre el uso.
  * 
  * @param aArgs - <tt>aArgs[0]</tt> es el nombre completo de un directorio 
  * existente que puede ser leído.
  */
  public static void main(String... aArgs) throws FileNotFoundException {
    File startingDirectory= new File(aArgs[0]);
    FileListing listing = new FileListing();
    List<File> files = listing.getFileListing(startingDirectory);

    //imprima todos los nombres de archivo, en el orden de File.compareTo ()
    for(File file : files){
      System.out.println(file);
    }
  }
  
  /**
  * Camine de forma recursiva por un árbol de directorios y devuelva una lista de todos
  * los Files encontrados; la List está ordenada usando File.compareTo().
  *
  * @param aStartingDir es un directorio válido, que puede ser leído.
  */
  public List<File> getFileListing(
    File aStartingDir
  ) throws FileNotFoundException {
    validateDirectory(aStartingDir);
    List<File> result = getFileListingNoSort(aStartingDir);
    Collections.sort(result);
    return result;
  }

  // PRIVADO

  private List<File> getFileListingNoSort(
    File aStartingDir
  ) throws FileNotFoundException {
    List<File> result = new ArrayList<>();
    File[] filesAndDirs = aStartingDir.listFiles();
    List<File> filesDirs = Arrays.asList(filesAndDirs);
    for(File file : filesDirs) {
      result.add(file); //siempre agregue, incluso si el directorio
      if (! file.isFile()) {
        //debe ser un directorio
        //¡llamada recursiva!
        List<File> deeperList = getFileListingNoSort(file);
        result.addAll(deeperList);
      }
    }
    return result;
  }

  /**
  * Directory es válido si este existe, no representa un archivo, y puede ser leído.
  */
  private void validateDirectory (
    File aDirectory
  ) throws FileNotFoundException {
    if (aDirectory == null) {
      throw new IllegalArgumentException("El directorio no debe ser nulo.");
    }
    if (!aDirectory.exists()) {
      throw new FileNotFoundException("El directorio no existe: " + aDirectory);
    }
    if (!aDirectory.isDirectory()) {
      throw new IllegalArgumentException("No es un directorio: " + aDirectory);
    }
    if (!aDirectory.canRead()) {
      throw new IllegalArgumentException("El directorio no se puede leer: " + aDirectory);
    }
  }
}
