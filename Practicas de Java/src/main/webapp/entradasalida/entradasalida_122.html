El buffering es generalmente apropiado

Las clases de entrada/salida sin búfer operan solo con un byte a la vez, mientras que los flujos con búfer interactúan con el disco en trozos de mayor tamaño. La entrada/salida con búfer se recomienda usualmente, ya que es generalmente más rápida, y probablemente debería ser el estilo predeterminado.

Con ficheros pequeños, el almacenamiento en búfer generalmente hace poca o ninguna diferencia. Sin embargo, puede ser difícil predecir si un fichero permanecerá siempre pequeño.

Puede ejecutar la siguiente clase de demostración en su equipo, para buscar diferencias entre la entrada con búfer y sin búfer. Con ficheros más grandes (~1 Meg), podría ver una diferencia de 2-3 veces. (¡En Rendimiento de la Plataforma Java por Wilson y Kesselman, un ejemplo usando un fichero JPEG de 370K tiene una ganancia de velocidad de ejecución de 83x!)

import java.io.*;

/** KDJ 6- */
public final class BufferDemo {

  public static void main (String... aArguments) {
    File file = new File("C:\\Temp\\blah.txt");
    verifyFile( file );

    Stopwatch stopwatch = new Stopwatch();

    stopwatch.start();
    readWithBuffer(file);
    stopwatch.stop();
    System.out.println("Con almacenamiento en búfer: " + stopwatch);

    stopwatch.start();
    readWithoutBuffer(file);
    stopwatch.stop();
    System.out.println("Sin almacenamiento en búfer: " + stopwatch);
  }

  /**
  * @param aFile es un fichero que ya existe y puede ser leído.
  */
  static public void readWithBuffer(File aFile) {
    try { 
      //utiliza almacenamiento en búfer, con un tamaño predeterminado de 8K
      Reader input =  new BufferedReader(new FileReader(aFile));
      try {
        int data = 0;
        while ((data = input.read()) != -1){
          //no hace nada
        }
      } 
      finally {
         input.close();
      }
    }
    catch (IOException ex){
      ex.printStackTrace();
    }
  }

  /**
  * @param aFile es un fichero existente que puede ser leído.
  */
  static public void readWithoutBuffer(File aFile) {
    try {
      Reader input =  new FileReader(aFile);
      try {
        //no usa almacenamiento en búfer
        int data = 0;
        while ((data = input.read()) != -1){
          //no hace nada
        }
      }
      finally {
        input.close();
      }
   }
   catch (IOException ex){
     ex.printStackTrace();
   }
 }

  private static void verifyFile(File aFile) {
    if (aFile == null) {
      throw new IllegalArgumentException("El archivo no debe ser nulo.");
    }
    if (!aFile.exists()) {
      throw new IllegalArgumentException ("El fichero no existe: " + aFile);
    }
    if (!aFile.isFile()) {
      throw new IllegalArgumentException("No debería ser un directorio: " + aFile);
    }
    if (!aFile.canWrite()) {
      throw new IllegalArgumentException("El archivo no se puede escribir: " + aFile);
    }
  }
}
