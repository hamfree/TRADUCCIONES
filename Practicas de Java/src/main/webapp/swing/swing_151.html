Verificar la entrada con expresiones regulares

Los objetos JTextField son a menudo usados para permitir la entrada del usuario y la edición de los datos. La clase InputVerifier se usa para verificar dichas entradas.

Example

RegexInputVerifier usa expresiones regulares para verificar la entrada. La verificación se realiza cuando el foco se mueve de un JTextField asociado; si se encuentra un error, el sistema emite un pitido y se coloca un mensaje de error en el JTextField.

Elementos a tener en cuenta:

  • shouldYieldFocus y verify se definieron originalmente en InputVerifier
  • el constructor usa una enumeración segura (UseToolTip) como un argumento, en vez de un valor primitivo boolean. Esto fuerce las llamadas de constructor para tener mayor claridad.
  • hay varios objetos estáticos de conveniencia proporcionados al final de esta clase, lo cual es útil para muchos casos comunes 

/**
* Verifica la entrada de usuario en un {@link javax.swing.text.JTextComponent} contra una 
* expresión regular.
* 
*<P> Tras la detección de una entrada no válida, esta clase realiza las siguientes acciones: 
*<ul>
* <li> emitir un sonido
* <li> sobreescribir el <tt>JTextComponent</tt> para visualizar lo siguiente: 
* INVÁLIDO: " (datos de entrada) "
* <li> opcionalmente, agregar el texto de información flotante al contenido del mensaje INVÁLIDO; esto 
* es útil solo si la información flotante contiene información útil referente a la entrada.
* Advertencia: agregar el texto de información flotante puede causar el error de
* texto muy largo para el campo de texto correspondiente. 
*</ul>
*
*<P> Se anima al usuario de esta clase a que siempre ponga 
* condiciones en la entrada de datos en la información sobre 
* herramientas para el campo correspondiente.
*/
final class RegexInputVerifier extends InputVerifier {
  
  /**
  * Constructor.
  *  
  * @param aPattern expresión regular contra las que todas las entradas de usuario serán 
  * verificadas; <tt>aPattern.pattern</tt> satisface 
  * {@link Util#textHasContent}.
  * @param aUseToolTip indica si el texto de información flotante deben ser agregadas a 
  * los mensajes de error visualizadas al usuario. 
  */
  RegexInputVerifier(Pattern aPattern, UseToolTip aUseToolTip){
    Args.checkForContent( aPattern.pattern() );
    fMatcher = aPattern.matcher(Consts.EMPTY_STRING);
    fUseToolTip = aUseToolTip.getValue();
  }
  
  /** La enumeración obliga a la persona que llama a usar un estilo que se lea claramente. */
  enum UseToolTip {
    TRUE(true),
    FALSE(false);
    boolean getValue(){
      return fToggle;
    }
    private boolean fToggle;
    private UseToolTip(boolean aToggle){
      fToggle = aToggle;
    }
  }

  /**
  * Siempre devuelve <tt>true</tt>, en esta implementación, de forma que el foco pueda 
  * siempre transferir a otro componente cuando la validación falla.
  *
  * <P>Si <tt>super.shouldYieldFocus</tt> devuelve  <tt>false</tt>, entonces 
  * notifica al usuario de un error.
  *
  * @param aComponent es un <tt>JTextComponent</tt>.
  */
  @Override public boolean shouldYieldFocus(JComponent aComponent){
    boolean isValid = super.shouldYieldFocus(aComponent);
    if (isValid){
      //no hace nada
    }
    else {
      JTextComponent textComponent = (JTextComponent)aComponent;
      notifyUserOfError(textComponent);
    }
    return true;
  }
  
  /**
  * Devuelve <tt>true</tt> solo si la entrada del usuario sin recortar coincide con la
  * expresión regular proporcionada al constructor.
  *
  * @param aComponent debe ser un <tt>JTextComponent</tt>.
  */
  @Override public boolean verify(JComponent aComponent) {
    boolean result = false;
    JTextComponent textComponent = (JTextComponent)aComponent;
    fMatcher.reset(textComponent.getText());
    if (fMatcher.matches()) {
      result =  true;
    }
    return result;
  }
  
  /**
  * El texto con el que empiezan todos los mensajes de error.
  *
  * La función llamadora puede examinar sus campos de textos por la presencia de 
  * <tt>ERROR_MESSAGE_START</tt>, antes de procesar la entrada.
  */
  static final String ERROR_MESSAGE_START = "INVÁLIDO: ";

  /**
  * Compara la entrada del usuario con una expresión regular.
  */
  private Matcher fMatcher;
  
  /**
  * Indica si el texto de información flotante de JTextField se debe agregar
  * a los mensajes de error, como una segunda forma de recordarle al usuario.
  */
  private boolean fUseToolTip;

  /* 
  * Varios patrones de expresiones regulares utilizados para 
  * construir objetos de conveniencia de esta clase:
  */
  
  private static final String TEXT_FIELD =  "^(//S)(.){1,75}(//S)$";
  private static final String NON_NEGATIVE_INTEGER_FIELD = "(//d){1,9}";
  private static final String INTEGER_FIELD = "(-)?" + NON_NEGATIVE_INTEGER_FIELD;
  private static final String NON_NEGATIVE_FLOATING_POINT_FIELD = 
    "(//d){1,10}//.(//d){1,10}"
  ;
  private static final String FLOATING_POINT_FIELD =  
    "(-)?" + NON_NEGATIVE_FLOATING_POINT_FIELD
  ;
  private static final String NON_NEGATIVE_MONEY_FIELD =  "(//d){1,15}(//.(//d){2})?";
  private static final String MONEY_FIELD =  "(-)?" + NON_NEGATIVE_MONEY_FIELD;
  
  /**  
  * Objeto de conveniencia para la entrada de enteros: ...-2,-1,0,1,2...
  *
  * <P>Desde 1 a 9 dígitos, posiblemente precedido por un signo menos.
  * Corresponde aproximadamente a la especificación de <tt>Integer.parseInt</tt>.
  * El límite en el número de dígitos se relaciona al tamaño de <tt>Integer.MAX_VALUE</tt>
  * and <tt>Integer.MIN_VALUE</tt>.
  */
  static final RegexInputVerifier INTEGER = 
    new RegexInputVerifier(Pattern.compile(INTEGER_FIELD), UseToolTip.FALSE)
  ;

  /**
  * Objeto de conveniencia para la entrada de estos enteros: 0,1,2...
  *
  *<P> Como en {@link #INTEGER}, pero sin signo menos inicial.
  */
  static final RegexInputVerifier NON_NEGATIVE_INTEGER = 
    new RegexInputVerifier(Pattern.compile(NON_NEGATIVE_INTEGER_FIELD), UseToolTip.FALSE)
  ;
         
  /**
  * Objeto de conveniencia para la introducción de pequeñas cantidades de texto.
  *
  * <P>Texto que contiene de 1 a 75 caracteres que no son espacios.
  */
  static final RegexInputVerifier TEXT = 
    new RegexInputVerifier(Pattern.compile(TEXT_FIELD), UseToolTip.FALSE)
  ;
    
  /**
  * Objeto de conveniencia para la entrada de números decimales, ej -23.23321, 100.25.
  *
  * <P>Posible signo menos inicial, de 1 a 10 dígitos antes del decimal y de 1 a 10 
  * dígitos después del decimal.
  */
  static final RegexInputVerifier FLOATING_POINT = 
    new RegexInputVerifier(Pattern.compile(FLOATING_POINT_FIELD), UseToolTip.FALSE)
  ;

  /**
  * Objeto de conveniencia para la entrada de números decimales no negativos, ej 23.23321, 100.25.
  *
  * <P>Como en {@link #FLOATING_POINT}, pero sin signo menos inicial.
  */
  static final RegexInputVerifier NON_NEGATIVE_FLOATING_POINT = 
    new RegexInputVerifier(
      Pattern.compile(NON_NEGATIVE_FLOATING_POINT_FIELD), UseToolTip.FALSE
    )
  ;

  /**
  * Objeto de conveniencia para la entrada de valores de dinero, ej -23, 100.25.
  *
  * <P>Posible signo menos inicial, de 1 a 15 dígitos iniciales y, opcionalmente, 
  * un decimal y dos decimales.
  */
  static final RegexInputVerifier MONEY = 
    new RegexInputVerifier(Pattern.compile(MONEY_FIELD), UseToolTip.FALSE)
  ;
  
  /**
  * Objeto de conveniencia para la entrada de valores de dinero no negativos, ej 23, 100.25.
  *
  * <P>Como en {@link #MONEY}, excepto sin signo menos inicial
  */
  static final RegexInputVerifier NON_NEGATIVE_MONEY = 
    new RegexInputVerifier(Pattern.compile(NON_NEGATIVE_MONEY_FIELD), UseToolTip.FALSE)
  ;

  /**
  * Si un mensaje de error es actualmente visualizado en aComponent, entonces 
  * no hace nada; de otra forma, visualiza un mensaje de error al usuario en un 
  * aComponent (vea la descripción de la clase para el formato del mensaje).
  */
  private void notifyUserOfError(JTextComponent aTextComponent){
    if ( isShowingErrorMessage(aTextComponent) ){
      //no hace nada, dado que el usuario aún no ha vuelto a ingresar.
    }
    else {
      UiUtil.beep();
      showErrorMessage(aTextComponent);
    }
  }

  private boolean isShowingErrorMessage(JTextComponent aTextComponent){
    return aTextComponent.getText().startsWith(ERROR_MESSAGE_START);
  }
  
  private void showErrorMessage(JTextComponent aTextComponent) {
    StringBuilder message = new StringBuilder(ERROR_MESSAGE_START);
    message.append("/"");
    message.append(aTextComponent.getText());
    message.append("/"");
    if ( fUseToolTip ) {
      message.append(aTextComponent.getToolTipText());
    }
    aTextComponent.setText(message.toString());
  }
 
  //omitido...
}
