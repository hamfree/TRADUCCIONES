Pantalla de Bienvenida

Las Pantallas de bienvenida son gráficos simples que se cargan rápidamente durante el arranque, y asegurarle al usuario que la aplicación se está cargando rápidamente. La clase SplashScreen fue agregada a JSE 6, y debe usarse si está disponible.

Si JSE 6 no está disponible, entonces la técnica siguiente se proporciona como alternativa.

Son posibles varias políticas para controlar la aparición y desaparición de la pantalla de bienvenida. Aquí, la pantalla de bienvenida desaparece cuando una ventana principal se ha terminado de cargar.

Este ejemplo también es de interés ya que demuestra claramente que el subproceso de lanzamiento es inusual - es distinto del subproceso de despacho de eventos, donde tiene lugar la mayor parte del trabajo de una aplicación Swing:

  • al iniciarse, el subproceso de lanzamiento muestra una pantalla de bienvenida, que activa el subproceso de despacho de eventos
  • una vez se muestra la pantalla de bienvenida (o, más precisamente, se "realiza"), el subproceso de lanzamiento actúa como un "subproceso de trabajo", y puede solo cerrar la pantalla de bienvenida indirectamente, usando EventQueue.invokeLater para inyectar correctamente un evento en el hilo de envío de eventos 

package hirondelle.stocks.main;

import java.awt.*;
import java.net.URL;
import java.awt.image.ImageObserver;

/**
* Presenta un gráfico simple al usuario hasta el lanzamiento de la aplicación, para 
* proporcionar una respuesta inicial más rápida que la que es posible con 
* la ventana principal.
* 
* <P>Adaptado de un 
* <a href=http://developer.java.sun.com/developer/qow/archive/24/index.html>elemento</a> 
* en Sun's Java Developer Connection.
*
* <P>Esta pantalla de bienvenida aparece en unos 2.5 segundos en una máquina 
* de desarrollo. La pantalla principal toma alrededor de 6.0 segundos en cargarse, así que el uso de la pantalla 
* de bienvenida reduce el retraso de visualización inicial en aproximadamente un 55 por ciento.
* 
* <P>Cuando el KDJ 6+ está disponible, se debe usar su clase java.awt.SplashScreen en vez 
* de esta clase.
*/
final class SplashScreen extends Frame {

  /**
  * Construya usando una imagen para la pantalla de presentación.
  *  
  * @param aImageId debe tener contenido, y se usa por 
  * {@link Class#getResource(java.lang.String)} para recuperar la imagen de la pantalla de bienvenida.
  */
  SplashScreen(String aImageId) {
    /* 
    * Nota de Implementación
    * Args.checkForContent no se llama aquí, en un intento de minimizar 
    * la carga de la clase.
    */
    if (aImageId == null || aImageId.trim().length() == 0){
      throw new IllegalArgumentException("El ID de la imagen no tiene contenido.");
    }
    fImageId = aImageId;
  }
   
  /**
  * Muestra la pantalla de presentación al usuario final.
  *
  * <P>Una vez que este método regresa, se realiza la pantalla de presentación, lo que significa 
  * que casi todo el trabajo en la pantalla de presentación debe continuar a través del hilo de envío 
  * de eventos. En particular, cualquier llamada a <tt>dispose</tt> para la 
  * pantalla de presentación debe realizarse en el subproceso de envío de eventos.
  */
  void splash(){
    initImageAndTracker();
    setSize(fImage.getWidth(NO_OBSERVER), fImage.getHeight(NO_OBSERVER));
    center();
    fMediaTracker.addImage(fImage, IMAGE_ID);
    try {
      fMediaTracker.waitForID(IMAGE_ID);
    }
    catch(InterruptedException ex){
      System.out.println("Cannot track image load.");
    }
    SplashWindow splashWindow = new SplashWindow(this,fImage);
  }
  
  // PRIVADO
  private final String fImageId;
  private MediaTracker fMediaTracker;
  private Image fImage;
  private static final ImageObserver NO_OBSERVER = null; 
  private static final int IMAGE_ID = 0;

  private void initImageAndTracker(){
    fMediaTracker = new MediaTracker(this);
    URL imageURL = SplashScreen.class.getResource(fImageId);
    fImage = Toolkit.getDefaultToolkit().getImage(imageURL);
  }

  /**
  * Centra el marco en la pantalla.
  *
  *<P>Este servicio de centrado está más o menos en {@link hirondelle.stocks.util.ui.UiUtil}; 
  * esta duplicación se justifica solo porque el uso de 
  * {@link hirondelle.stocks.util.ui.UiUtil} implicaría más carga de clases, lo que no es 
  * deseable para una pantalla de presentación.
  */
  private void center(){
    Dimension screen = Toolkit.getDefaultToolkit().getScreenSize();
    Rectangle frame = getBounds();
    setLocation((screen.width - frame.width)/2, (screen.height - frame.height)/2);
  }
 
  private final class SplashWindow extends Window {
    SplashWindow(Frame aParent, Image aImage) {
       super(aParent);
       fImage = aImage;
       setSize(fImage.getWidth(NO_OBSERVER), fImage.getHeight(NO_OBSERVER));
       Dimension screen = Toolkit.getDefaultToolkit().getScreenSize();
       Rectangle window = getBounds();
       setLocation((screen.width - window.width) / 2,(screen.height - window.height)/2);
       setVisible(true);
    }
    @Override public void paint(Graphics graphics) {
      if (fImage != null) {
        graphics.drawImage(fImage,0,0,this);
      }
    }
    private Image fImage;
  }
  
  /**
  * La prueba del desarrollador muestra la pantalla de inicio durante un período de tiempo fijo, 
  * sin iniciar la aplicación completa.
  */
  private static void main(String... aArgs){
    SplashScreen splashScreen = new SplashScreen("StocksMonitor.gif");
    splashScreen.splash();
    try {
      Thread.sleep(2000);
    }
    catch(InterruptedException ex) {
      System.out.println(ex);
    }
    System.exit(0);
  }
}

Aquí tiene un ejemplo de una clase que lanza una aplicación usando SplashScreen: 
package hirondelle.stocks.main;

import java.util.logging.*;
import java.awt.EventQueue;
import hirondelle.stocks.util.Util;
import hirondelle.stocks.util.Consts;

/**
* Lanza la aplicación usando una versión más antigua de la pantalla de bienvenida.
*
*<P>Realiza las tareas en este orden:
*<ul>
* <li>registra información básica del sistema 
* <li>muestra rápidamente una pantalla de bienvenida al inicio
* <li>muestra la pantalla principal
* <li>elimina la pantalla de bienvenida una vez que se muestra la pantalla principal
*</ul>
*
* Estas tareas se realizan de forma segura para subprocesos.
*/
public final class Launcher { 

  /**
  * Inicia la aplicación y muestra la ventana principal.
  *
  * @param aArgs son ignorados por esta aplicación, y pueden tomar cualquier valor.
  */
  public static void main (String... aArgs) {
    
    /*
    * Nota de implementación:
    *
    * Tenga en cuenta que el hilo de inicio de cualquier aplicación GUI es, en efecto, 
    * un hilo de trabajo inicial; no es el hilo de despacho de eventos, donde tiene lugar 
    * la mayor parte del procesamiento. Por lo tanto, una vez que el subproceso de lanzamiento se da cuenta de una ventana, 
    * entonces el subproceso de lanzamiento casi siempre debería manipular dicha ventana a través 
    * de EventQueue.invokeLater. (Esto se hace para cerrar la pantalla de 
    * bienvenida, por ejemplo).
    */
    
    //verifica que las afirmaciones estén en:
    //  assert(false) : "Test";
    
    logBasicSystemInfo();
    showSplashScreen();
    showMainWindow();
    EventQueue.invokeLater(new SplashScreenCloser());
    fLogger.info("Subproceso de de lanzamiento ahora saliendo...");
  }

  // PRIVADO
  
  private static SplashScreen fSplashScreen;
  private static final Logger fLogger = Util.getLogger(Launcher.class);
  private static final String SPLASH_IMAGE = "StocksMonitor.gif";

  /**
  * Muestra una pantalla de presentación gráfica simple, como preliminar rápido a la pantalla principal.
  */
  private static void showSplashScreen(){
    fLogger.info("Mostrando la pantalla de bienvenida.");
    fSplashScreen = new SplashScreen(SPLASH_IMAGE);
    fSplashScreen.splash();
  }
  
  /**
  * Muestra la ventana principal de la aplicación al usuario.
  */
  private static void showMainWindow(){
    fLogger.info("Mostrando la pantalla principal.");
    StocksMonitorMainWindow mainWindow = new StocksMonitorMainWindow();
  }

  /** 
  * Elimina la pantalla de presentación. 
  *
  * Invoca esta <tt>Runnable</tt> usando 
  * <tt>EventQueue.invokeLater</tt>, para eliminar la pantalla de bienvenido
  * de una manera segura para el procesado multiproceso.
  */
  private static final class SplashScreenCloser implements Runnable {
    @Override public void run(){
      fLogger.fine("Cerrando la pantalla de presentación.'");
      fSplashScreen.dispose();
    }
  }
  
  private static void logBasicSystemInfo() {
    fLogger.info("Lanzando la aplicación...");
    fLogger.config(
      "Operating System: " + System.getProperty("os.name") + " " + 
      System.getProperty("os.version")
    );
    fLogger.config("EEJ: " + System.getProperty("java.version"));
    fLogger.info("Java Lanzado Desde: " + System.getProperty("java.home"));
    fLogger.config("Ruta de Clases: " + System.getProperty("java.class.path"));
    fLogger.config("Ruta de Librerías: " + System.getProperty("java.library.path"));
    fLogger.config("Nombre de la Aplicación: " + Consts.APP_NAME + "/" + Consts.APP_VERSION);
    fLogger.config("Directorio Inicial del Usuario: " + System.getProperty("user.home"));
    fLogger.config("Directorio de Trabajo del Usuario: " + System.getProperty("user.dir"));
    fLogger.info("Prueba registro INFO.");
    fLogger.fine("Prueba de registro FINE.");
    fLogger.finest("Prueba el registro FINEST.");
  }
}
